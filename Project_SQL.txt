

CREATE TABLE "User" ("User_ID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE  DEFAULT 1,
					"Password" VARCHAR NOT NULL  DEFAULT "",
					"First_Name" VARCHAR NOT NULL  DEFAULT "",
					"Last_Name" VARCHAR NOT NULL  DEFAULT "",
					"Email" VARCHAR NOT NULL  DEFAULT "",
					"Batch_Num" INTEGER NOT NULL  DEFAULT -1,
					"Num_Record" INTEGER NOT NULL  DEFAULT -1,
					 "User_Name" VARCHAR NOT NULL  UNIQUE DEFAULT "");


CREATE TABLE "Project" ("Project_ID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  DEFAULT (-1) ,
						"Project_Title" VARCHAR NOT NULL  DEFAULT ('') ,
						"Batch_Count" INTEGER NOT NULL  DEFAULT (-1) ,
						"First_Y_Coord" INTEGER NOT NULL  DEFAULT (-1) ,
						"Records/batch" INTEGER NOT NULL  DEFAULT (-1) ,
						"Record_Height" INTEGER NOT NULL  DEFAULT (-1) );

CREATE TABLE "Batch" ("Batch_ID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE  DEFAULT -1, 
					"Project_ID" INTEGER NOT NULL  DEFAULT -1, 
					"Checked_out" BOOL NOT NULL  DEFAULT 0, 
					"Finished" BOOL NOT NULL  DEFAULT 0,
					"Image_Path" VARCHAR NOT NULL  DEFAULT "");


CREATE TABLE "Record" ("Record_ID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE  DEFAULT -1,
					   "Batch_ID" INTEGER NOT NULL  UNIQUE  DEFAULT -1);

CREATE TABLE "Field" ("Field_ID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE  DEFAULT -1,
					"Field_Name" VARCHAR NOT NULL  DEFAULT "",
					"Project_ID" INTEGER NOT NULL  UNIQUE  DEFAULT -1,
					"Column_Num" INTEGER NOT NULL  DEFAULT 0,
					"First_X_Coord" INTEGER NOT NULL  DEFAULT -1,
					"Width" INTEGER NOT NULL  DEFAULT -1,
					"Known_Data_Path" VARCHAR NOT NULL  DEFAULT "",
					"Field_Help_Path" VARCHAR NOT NULL  DEFAULT "");


CREATE TABLE "Value" ("Value_ID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE  DEFAULT -1,
					"Value_Name" VARCHAR NOT NULL  DEFAULT "",
					"Record_ID" INTEGER NOT NULL  UNIQUE  DEFAULT -1,
					"Column_Num" INTEGER NOT NULL  DEFAULT 0);
